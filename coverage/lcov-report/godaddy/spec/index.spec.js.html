'use strict';
 
var assert = require('assert');
var godaddy = require('../index.js');
var config = require('../config');
var lang = require('../lang');
var nock = require('nock');
var lodash = require('lodash');
var nockurl = nock(`${config.url}`);
var nockurlAuth = nock(`${config.url}`, {
  reqheaders: {
    Authorization: `sso-key ${config.key}:${config.secrect}`
  }
});
 
function ErrorHandler (err) {
  let erro = new Error('Error: ' + err);
  throw erro;
}
 
describe('ErrorHandler', () => {
  it('ErrorHandler()', () => {
    try {
      ErrorHandler('teste');  
    } catch(e) {
      assert.ok(e, 'Error: testte');
    }
  });
})
 
describe('Godaddy', () => {
 
  describe('Domain', () => {
 
    it('#suggest()', (done) => {
      let response = [
        { domain: 'undefinedweb.com.br' },
        { domain: 'undefinedpro.com.br' },
        { domain: 'undefined.com.br' },
        { domain: 'undefinedhub.com.br' },
        { domain: 'undefinedshop.com.br' }
      ];
 
      nockurl
        .get('/domains/suggest?query=teste&tlds=com.br')
        .reply(200, response);
 
      godaddy
        .Domain
        .suggest('teste')
        .then((e) => {
          let reponseIsEqual = lodash.isEqual(e, response);
 
          assert.ok(reponseIsEqual);
          
          done();
        })
        .catch(ErrorHandler);
    });
 
    it('#available()', (done) => {
      let response = {
        available: true
      }
 
      nockurl
        .get('/domains/available?domain=noname.com')
        .reply(200, response);
      
      godaddy
        .Domain
        .available('noname.com')
        .then((isAvailable) => {
 
          done(assert.strictEqual(isAvailable, true));
        })
        .catch(ErrorHandler);
    });
 
    it('create new Domain', () => {
      const url = 'a.noname.com';
      let domain = new godaddy.Domain(url);
 
      assert.equal(domain.domain, url);
    });
 
    it('#addCname()', (done) => {
      let url = 'a.noname.com';
      let name = 'b';
      let domain = new godaddy.Domain(url);
      
      nockurlAuth
        .put(`/domains/${url}/records/CNAME/${name}`)
        .reply(200, {});
 
      domain
        .addCname('b', 'a.noname.com')
        .then((response) => {
          assert.equal(response.message, `${lang["message success addCname"]}`);
          done();
        })
        .catch(ErrorHandler);
    });
 
    it('#purchase()', (done) => {
      done(assert.ok(false));
    });
  });
});
