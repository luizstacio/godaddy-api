'use strict';
 
var request = require('request-promise');
var config = require('./config');
var util = require('./util');
var lang = require('./lang');
 
class Domain {
  constructor(domain) {
    this.domain = domain;
  }
 
  addCname (name, data) {
    let req = util.applyAuth({    
      method: 'PUT',
      url: `${config.url}/domains/${this.domain}/records/CNAME/${name}`,
      body: [{ data }],
      json: true
    });
 
    let promise = new Promise((done, reject) => {
      request(req).then((reponse) => {
        done({ message: `${lang["message success addCname"]}` });
      }).catch(reject);
    });
 
    return promise;
  }
 
  static suggest (query) {
    let req = util.applyAuth({
      method: 'GET',
      url: `${config.url}/domains/suggest?query=${query}&tlds=${config.tlds}`,
      json: true
    });
 
    return request(req);
  }
 
  static available (domain) {
    let req = util.applyAuth({
      method: 'GET',
      url: `${config.url}/domains/available?domain=${domain}`,
      json: true
    });
    let promise = new Promise((done, reject) => {
      request(req).then((reponse) => {
        done(reponse.available);
      }).catch(reject);
    })
 
    return promise;
  }
}
 
module.exports = {
  Domain
}
